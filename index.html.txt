    <h1>البحث عن نتيجة الطالب</h1>

    <input type="text" id="search" placeholder="ادخل الاسم أو رقم الجلوس">
    <button onclick="searchResult()">بحث</button>

    <div id="result"></div>

    <script>
        let students = [];

        fetch('result.csv')
            .then(response => response.text())
            .then(data => {
                students = data.split('\n').slice(1).map(line => line.split(','));
            });

        function searchResult() {
            let query = document.getElementById('search').value.trim();
            let resultArea = document.getElementById('result');
            let found = students.filter(s => s[0] === query || s[1].includes(query));

            if (found.length > 0) {
                let table = '<table><tr><th>رقم الجلوس</th><th>الاسم</th><th>المجموع</th></tr>';
                found.forEach(s => {
                    table += `<tr><td>${s[0]}</td><td>${s[1]}</td><td>${s[2]}</td></tr>`;
                });
                table += '</table>';
                resultArea.innerHTML = table;
            } else {
                resultArea.innerHTML = '<p>لم يتم العثور على الطالب</p>';
            }
        }
    </script>
